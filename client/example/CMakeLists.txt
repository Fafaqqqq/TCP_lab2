cmake_minimum_required(VERSION 3.21)

project(${target})

    src/*.h
    src/*.cpp
)

add_executable(${target} ${sources})

set_target_properties(${target} PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/bin)

target_link_libraries(${target})
